  1. 先将数组排序，然后遍历不等于当前元素就push进新的数组
let m = [];
   arr.sort();
   m.push(arr[0]);
   for (let i = 1; i < arr.length; i++) {
      if (arr[i] != m[m.length - 1]) {
        m.push(arr[i]);
      }
   }

3.通过array.indexOf()方法

   let arr = [1, 1, 2, 2, 7, 5, 6, 6]
   let m = []
   for (let i = 0; i<arr.length; i++) {
		if(m.indexOf(arr[i])<0) {
         m.push(arr[i])
      }
   }

4.通过es6的Map对象

   let arr = [1, 1, 2, 2, 7, 5, 6, 6]
   let map = new Map()
   let res = arr.filter(item => {
      if(!map.has(item)) {
         map.set(item)
         return map
      }
   })
